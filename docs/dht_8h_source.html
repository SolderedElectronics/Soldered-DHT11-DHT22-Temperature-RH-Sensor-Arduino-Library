40// max timeout is 100 usec.
41// For a 16 Mhz proc 100 usec is 1600 clock cycles
42// loops using DHTLIB_TIMEOUT use at least 4 clock cycli
43// so 100 us takes max 400 loops
44// so by dividing F_CPU by 40000 we "fail" as fast as possible
45#ifndef F_CPU
46#define DHTLIB_TIMEOUT 1000 // ahould be approx. clock/40000
